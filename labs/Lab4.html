Laboratory Task 4#

Name: Karylle B. dela Cruz
Section: DS4A

Instruction: Train a linear regression model in PyTorch using a regression dataset. Use the following parameters.

  • Criterion: MSE Loss

  • Fully Connected Layers x 2

  • Batch Size: 8

  • Optimizer: SGD

  • Epoch: 1000

import torch
import numpy as np
import sys
import torch
import torch.nn as nn
import torch.optim as optim
from torch.utils.data import DataLoader, TensorDataset

device = torch.device("cuda:0" if (torch.cuda.is_available()) else "cpu")
print("Device: ", device)

Device:  cpu

  1. Specifiying input and target

# Input (temp, rainfall, humidity)
x_train = np.array([
    [73, 67, 43], [91, 88, 64], [87, 134, 58],

    [56, 70], [81, 101], [119, 133], 
    [22, 37], [103, 119], [56, 70], 
    [81, 101], [119, 133], [22, 37],

    [81, 101], [119, 133], [22, 37], 
    [103, 119]], dtype='float32')
inputs = torch.from_numpy(x_train)
targets = torch.from_numpy(y_train)

  1. Wrap into TensorDataset + DataLoader (batch size = 8)

dataset = TensorDataset(inputs, targets)
train_dl = DataLoader(dataset, batch_size=8, shuffle=True)

  1. Define Model (Fully Connected Layers ×2)

model = nn.Sequential(
    nn.Linear(3, 16),   # Input 3 → Hidden 16
    nn.ReLU(),
    nn.Linear(16, 2)    # Hidden 16 → Output 2
).to(device)

  1. Loss and Optimizer

criterion = nn.MSELoss()
optimizer = optim.SGD(model.parameters(), lr=0.01)

  1. Training Loop

def fit(num_epochs, model, criterion, optimizer, train_dl):
    for epoch in range(num_epochs):
        total_loss = 0
        
        for xb, yb in train_dl:
            xb, yb = xb.to(device), yb.to(device)

            # 1. Forward pass
            pred = model(xb)
            loss = criterion(pred, yb)

            # 2. Backward pass
            optimizer.zero_grad()
            loss.backward()

            # 3. Update weights
            optimizer.step()

            total_loss += loss.item()
        
        # Print progress every 100 epochs
        if (epoch+1) % 100 == 0:
            avg_loss = total_loss / len(train_dl)
            print(f"Epoch [{epoch+1}/{num_epochs}], Loss: {avg_loss:.4f}")

  1. Train
